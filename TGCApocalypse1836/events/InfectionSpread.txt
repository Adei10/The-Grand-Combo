#Japan to China
country_event = {
	id = 66004
	title = "EVTNAME66002"
	desc = "EVTDESC66002"
	picture = emigration

	trigger = {
		tag = CHI
		JAP = {
			any_owned_province = {
				is_coastal = yes
				is_colonial = no
				is_overseas = no
				has_province_modifier = zombie_outbreak
			}
			has_country_flag = zombie_outbreak
		}
		NOT = {
			any_owned_province = {
				has_province_modifier = zombie_outbreak
			}
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
	}

	mean_time_to_happen = {
		months = 6
		modifier = { factor = 9 has_country_modifier = ap_lock_down }
	}

	option = {
		name = "Uh oh."
		set_country_flag = zombie_outbreak
		any_pop = {
			consciousness = 1
			militancy = -2
		}
		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = 5000
			}
			any_pop = {
				reduce_pop = 0.8
				consciousness = -3
				militancy = 10
			}
			JAN = { add_country_modifier = { name = "ap_zm_timer" duration = 5030 } }
		}
	}
}
