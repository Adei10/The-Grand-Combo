#A mysterious plague, outbreak
country_event = {
	id = 66001
	title = "EVTNAME66001"
	desc = "EVTDESC66001"
	picture = disease

	fire_only_once = yes
	
	trigger = {
		ai = no
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
	}

	mean_time_to_happen = {
		days = 10
	}

	immediate = {
		set_global_flag = zombie_mod
	}

	option = {
		name = "EVTOPTA66001"
		random_country = {
			limit = {
				civilized = yes
				is_vassal = no
				rank = 20
				number_of_states = 3
			}
			country_event = 66002
		}
	}

	option = {
		name = "No"
	}
}

#Country to Province
country_event = {
	id = 66002
	title = "EVTNAME66002"
	desc = "EVTDESC66002"
	picture = disease

	trigger = {
		any_neighbor_country = {
			has_country_flag = zombie_outbreak
		}
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
	}

	mean_time_to_happen = {
		months = 2
		modifier = { factor = 9 has_country_modifier = ap_lock_down }
	}

	option = {
		name = "EVTOPTA66002"

		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}
			militancy = 8
			add_province_modifier = { name = zombie_outbreak duration = 5000 }
			JAN = { add_country_modifier = { name = "ap_zm_timer" duration = 5030 } }
			any_pop = { consciousness = -2 }
		}
	}
}

	id = 66014
	title = "EVTNAME66014"
	desc = "EVTDESC66014"

	trigger = {
		has_province_modifier = zombie_outbreak
		NOT = {
			controlled_by = REB
			country_units_in_province = REB
		}
		units_in_province = 1
	}

	mean_time_to_happen = {
		months = 1
		modifier = {
			factor = 1.25
			any_neighbor_province = {
				has_province_modifier = zombie_outbreak
			}
		}
	}

	option = {
		name = "Excellent."
		remove_province_modifier = zombie_outbreak
		any_pop = {
			consciousness = 2
			militancy = -8
		}
	}
}
