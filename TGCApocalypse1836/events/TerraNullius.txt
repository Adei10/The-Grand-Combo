province_event = {      #Zombify province
	id = 66100
	title = "EVTNAME66025"
	desc = "EVTDESC66025"

	trigger = {
		owner = {exists = yes }
		NOT = { owner = { tag = ZOM } }
		province_control_days = 365
		has_province_modifier = zombie_outbreak
		controlled_by = REB
		exists = ZOM
	}

	mean_time_to_happen = {
		days = 10
	}

	immediate = { 
				JAN = {
					change_variable = {
							which = zombified_provinces
							value = 1
					}
				}
				any_pop = {
					reduce_pop = 0.8
					militancy = 8
					consciousness = -3
				}
				change_controller = ZOM
				secede_province = ZOM
				ZOM = {
					remove_country_modifier = do_not_annex
					add_country_modifier = {
						name = do_not_annex
						duration = 31
					}
				}
	}

	option = {
		name = "Damn!"
		owner = { prestige = -3 }
	}
}

country_event = { #note, this must be a country event, otherwise it might happen multiple times at once, which can cause a crash
	id = 66101
	title = "EVTNAME66101"
	desc = "EVTDESC66101"
	picture = plague

	trigger = {
		exists = yes
		NOT = { exists = ZOM }
		any_owned_province = {
			province_control_days = 365
			has_province_modifier = zombie_outbreak
			controlled_by = REB
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {		
		random_owned = {
			limit = {
				province_control_days = 365
				has_province_modifier = zombie_outbreak
				controlled_by = REB
			}
			any_pop = {
				reduce_pop = 0.4
				militancy = 8
				consciousness = -3
			}
			owner = { prestige = -3 }
			change_controller = ZOM
			secede_province = ZOM
		}
		ZOM = {
			remove_country_modifier = do_not_annex
			remove_country_modifier = accepts_provinces
			add_country_modifier = {
				name = do_not_annex
				duration = 5
			}
			country_event = { id = 66202 days = 31 }
		} 
	}

	option = {
		name = "Damn!"
	}
}
